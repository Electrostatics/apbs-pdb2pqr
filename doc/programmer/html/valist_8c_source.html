423 /* Get a the next available atom storage location, increasing the storage
424  * space if necessary. Return VNULL if something goes wrong. */
425 VPRIVATE Vatom* Valist_getAtomStorage(
426  Valist *thee,
427  Vatom **plist, /* Pointer to existing list of atoms */
428  int *pnlist, /* Size of existing list, may be changed */
429  int *pnatoms /* Existing number of atoms in list; incremented
430  before exit */
431  ) {
432 
433  Vatom *oldList, *newList, *theList;
434  Vatom *oldAtom, *newAtom;
435  int iatom, inext, oldLength, newLength, natoms;
436 
437  newList = VNULL;
438 
439  /* See if we need more space */
440  if (*pnatoms >= *pnlist) {
441 
442  /* Double the storage space */
443  oldLength = *pnlist;
444  newLength = 2*oldLength;
445  newList = (Vatom*)Vmem_malloc(thee->vmem, newLength, sizeof(Vatom));
446  oldList = *plist;
447 
448  /* Check the allocation */
449  if (newList == VNULL) {
450  Vnm_print(2, "Valist_readPDB: failed to allocate space for %d (Vatom)s!\n", newLength);
451  return VNULL;
452  }
453 
454  /* Copy the atoms over */
455  natoms = *pnatoms;
456  for (iatom=0; iatom<natoms; iatom++) {
457  oldAtom = &(oldList[iatom]);
458  newAtom = &(newList[iatom]);
459  Vatom_copyTo(oldAtom, newAtom);
460  Vatom_dtor2(oldAtom);
461  }
462 
463  /* Free the old list */
464  Vmem_free(thee->vmem, oldLength, sizeof(Vatom), (void **)plist);
465 
466  /* Copy new list to plist */
467  *plist = newList;
468  *pnlist = newLength;
469  }
470 
471  theList = *plist;
472  inext = *pnatoms;
473 
474  /* Get the next available spot and increment counters */
475  newAtom = &(theList[inext]);
476  *pnatoms = inext + 1;
477 
478  return newAtom;
479 }

519  * REQUIRE ALL FIELDS TO BE WHITESPACE DELIMITED */

590  } /* if ATOM or HETATM */
591  } /* while we haven't run out of tokens */

804 reading radius!\n", tok);

858 /* Load up Valist with various statistics */
859 VPUBLIC Vrc_Codes Valist_getStatistics(Valist *thee) {
860 
861  Vatom *atom;
862  int i, j;
863 
864  if (thee == VNULL) {
865  Vnm_print(2, "Valist_getStatistics: Got NULL pointer when loading up Valist with various statistics!\n");
866  VASSERT(0);
867  }
868 
869  thee->center[0] = 0.;
870  thee->center[1] = 0.;
871  thee->center[2] = 0.;
872  thee->maxrad = 0.;
873  thee->charge = 0.;
874 
875  if (thee->number == 0) return VRC_FAILURE;
876 
877  /* Reset stat variables */
878  atom = &(thee->atoms[0]);
879  for (i=0; i<3; i++) {
880  thee->maxcrd[i] = thee->mincrd[i] = atom->position[i];
881  }
882  thee->maxrad = atom->radius;
883  thee->charge = 0.0;
884 
885  for (i=0; i<thee->number; i++) {
886 
887  atom = &(thee->atoms[i]);
888  for (j=0; j<3; j++) {
889  if (atom->position[j] < thee->mincrd[j])
890  thee->mincrd[j] = atom->position[j];
891  if (atom->position[j] > thee->maxcrd[j])
892  thee->maxcrd[j] = atom->position[j];
893  }
894  if (atom->radius > thee->maxrad) thee->maxrad = atom->radius;
895  thee->charge = thee->charge + atom->charge;
896  }
897 
898  thee->center[0] = 0.5*(thee->maxcrd[0] + thee->mincrd[0]);
899  thee->center[1] = 0.5*(thee->maxcrd[1] + thee->mincrd[1]);
900  thee->center[2] = 0.5*(thee->maxcrd[2] + thee->mincrd[2]);
901 
902  Vnm_print(0, "Valist_getStatistics: Max atom coordinate: (%g, %g, %g)\n",
903  thee->maxcrd[0], thee->maxcrd[1], thee->maxcrd[2]);
904  Vnm_print(0, "Valist_getStatistics: Min atom coordinate: (%g, %g, %g)\n",
905  thee->mincrd[0], thee->mincrd[1], thee->mincrd[2]);
906  Vnm_print(0, "Valist_getStatistics: Molecule center: (%g, %g, %g)\n",
907  thee->center[0], thee->center[1], thee->center[2]);
908 
909  return VRC_SUCCESS;
910 }
